# MADE UP DATA

H13PR <- data.frame(age = rnorm(100, 20, 5),
                    s5 = c(rep("male", 50),
                           rep("female", 50)))

library(ggplot2)

# I COMMENTED SOME THINGS OUT TO TWEAK THE GGPLOT

ggplot(H13PR, 
       aes(x = as.numeric(age),
           fill = as.factor(s5))) +
  # geom_density(alpha = .5) + COMMENTED OUT
  geom_histogram() +
  # geom_dotplot(binwidth = 1.5) + COMMENTED OUT
  scale_fill_manual(name = "Gender",
                    values = c("blue", 
                               "pink"),
                    labels = c("Male",
                               "Female")) +
  labs(title = "Participant Age") +
  theme(axis.text.y = element_blank()) +
  facet_wrap(~s5)
